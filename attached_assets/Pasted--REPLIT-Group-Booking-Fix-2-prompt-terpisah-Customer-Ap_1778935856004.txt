## UPDATE `createClassBooking` di `lib/supabase.ts`

Cari fungsi `createClassBooking` — tambah `group_id` ke parameter dan INSERT:

```typescript
// Tambah group_id ke interface/parameter
export async function createClassBooking(params: {
  schedule_id: string
  booker_type: string
  customer_type: string
  full_name: string
  email: string
  phone: string
  notes?: string | null
  price: number
  price_before_disc: number
  discount: number
  group_id?: string | null    // ← TAMBAH INI
  status: string
}) {
  return supabase
    .from('arena_class_bookings')
    .insert({
      ...params,
      group_id: params.group_id ?? null,  // ← TAMBAH INI
    })
    .select('id, booking_code')
    .single()
}
```

### Komponen `GroupMembersInline`

```jsx
function GroupMembersInline({ groupId, currentId }) {
  const [members, setMembers] = useState([])

  useEffect(() => {
    supabase
      .from('arena_class_bookings')
      .select('id, booking_code, full_name, price, status')
      .eq('group_id', groupId)
      .neq('id', currentId)
      .then(({ data }) => setMembers(data || []))
  }, [groupId])

  if (members.length === 0) return null

  return (
    <div style={{ marginTop: 8 }}>
      <div style={{ fontSize: 11, color: 'var(--text-muted)', textTransform: 'uppercase', letterSpacing: '0.05em', marginBottom: 6 }}>
        Tiket lain dalam grup ini:
      </div>
      {members.map(m => (
        <div key={m.id} style={{
          display: 'flex', justifyContent: 'space-between',
          padding: '6px 0', borderTop: '1px solid var(--border)',
          fontSize: 13,
        }}>
          <span style={{ fontFamily: 'monospace', fontSize: 11 }}>{m.booking_code}</span>
          <span>{m.full_name}</span>
          <span style={{ color: 'var(--text-muted)' }}>{fmtRp(m.price)}</span>
          <span className={`badge badge-${m.status === 'confirmed' ? 'confirmed' : m.status === 'cancelled' ? 'cancelled' : 'pending'}`}>
            {m.status}
          </span>
        </div>
      ))}
    </div>
  )
}
```
