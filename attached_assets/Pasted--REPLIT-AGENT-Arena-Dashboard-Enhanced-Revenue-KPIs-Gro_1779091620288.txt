### Fetch semua data sekaligus

```javascript
const fetchDashboard = async (period) => {
  const { start, end } = getPeriodRange(period)

  const [classRes, slotIndRes, slotCorpRes] = await Promise.all([
    // Class bookings
    supabase
      .from('arena_class_bookings')
      .select('price, paid_at, status')
      .eq('status', 'confirmed')
      .gte('paid_at', start)
      .lte('paid_at', end),

    // Slot booking — individual only
    supabase
      .from('arena_bookings')
      .select('price, paid_at, status, customer_type')
      .eq('status', 'confirmed')
      .eq('customer_type', 'individual')
      .gte('paid_at', start)
      .lte('paid_at', end),

    // Venue rental — corporation only
    supabase
      .from('arena_bookings')
      .select('price, paid_at, status, customer_type, full_name')
      .eq('status', 'confirmed')
      .eq('customer_type', 'corporation')
      .gte('paid_at', start)
      .lte('paid_at', end),
  ])

  const sum = (rows) => (rows || []).reduce((s, r) => s + (Number(r.price) || 0), 0)

  const classRevenue = sum(classRes.data)
  const slotRevenue  = sum(slotIndRes.data)
  const venueRevenue = sum(slotCorpRes.data)
  const totalRevenue = classRevenue + slotRevenue + venueRevenue

  const totalCount = (classRes.data?.length || 0) + (slotIndRes.data?.length || 0) + (slotCorpRes.data?.length || 0)
  const avgPerBooking = totalCount > 0 ? Math.round(totalRevenue / totalCount) : 0

  setRevenue({ total: totalRevenue, class: classRevenue, slot: slotRevenue, venue: venueRevenue })
  setBookingCount(totalCount)
  setAvgPerBooking(avgPerBooking)

  // Top komunitas by venue revenue
  const venueByOrg = {}
  ;(slotCorpRes.data || []).forEach(r => {
    venueByOrg[r.full_name] = (venueByOrg[r.full_name] || 0) + Number(r.price)
  })
  const topKomunitas = Object.entries(venueByOrg)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
    .map(([name, amount]) => ({ name, amount }))
  setTopKomunitas(topKomunitas)
}

// Pending & Cancelled — tidak terikat period filter
const fetchStatusCounts = async () => {

    supabase.from('arena_class_bookings').select('id', { count: 'exact', head: true }).eq('status', 'cancelled'),
    supabase.from('arena_bookings').select('id', { count: 'exact', head: true }).eq('status', 'cancelled'),
  ])
  setPendingCount((pendingClass.count || 0) + (pendingSlot.count || 0))
  setCancelledCount((cancelClass.count || 0) + (cancelSlot.count || 0))
}
```

## ROW 3 — Charts

### Bar Chart 6 Bulan (3 bar per bulan)

Fetch data 6 bulan terakhir, aggregate per bulan per sumber:

```javascript
const fetchChartData = async () => {
  const months = Array.from({ length: 6 }, (_, i) => {
    const d = new Date()
    d.setMonth(d.getMonth() - (5 - i))
    return {
      label: d.toLocaleDateString('id-ID', { month: 'short' }),
      year: d.getFullYear(),
      month: d.getMonth() + 1,
      start: new Date(d.getFullYear(), d.getMonth(), 1).toISOString(),
      end: new Date(d.getFullYear(), d.getMonth() + 1, 0, 23, 59, 59).toISOString(),
    }
  })

  const chartData = await Promise.all(months.map(async (m) => {
    const [cls, slot, venue] = await Promise.all([
      supabase.from('arena_class_bookings').select('price').eq('status', 'confirmed').gte('paid_at', m.start).lte('paid_at', m.end),
      supabase.from('arena_bookings').select('price').eq('status', 'confirmed').eq('customer_type', 'individual').gte('paid_at', m.start).lte('paid_at', m.end),
      supabase.from('arena_bookings').select('price').eq('status', 'confirmed').eq('customer_type', 'corporation').gte('paid_at', m.start).lte('paid_at', m.end),
    ])
    const sum = (d) => (d.data || []).reduce((s, r) => s + Number(r.price), 0)
    return { label: m.label, class: sum(cls), slot: sum(slot), venue: sum(venue) }
  }))

  setChartData(chartData)
}
```

**Render bar chart (3 bar per bulan, native SVG/div):**

## ROW 4 — Booking Terbaru (paid_at DESC)

```javascript
const fetchRecentBookings = async () => {
  const [classRes, slotRes] = await Promise.all([
    supabase
      .from('arena_class_bookings')
      .select(`booking_code, full_name, price, status, paid_at,
        schedule:arena_class_schedules(schedule_date, class_type:arena_class_types(name))`)
      .eq('status', 'confirmed')
      .not('paid_at', 'is', null)
      .order('paid_at', { ascending: false })
      .limit(5),
    supabase
      .from('arena_bookings')
      .select('booking_code, full_name, price, status, paid_at, booking_date, customer_type')
      .eq('status', 'confirmed')
      .not('paid_at', 'is', null)
      .order('paid_at', { ascending: false })
      .limit(5),
  ])

  const all = [
    ...(classRes.data || []).map(r => ({ ...r, type: 'Class' })),
    ...(slotRes.data || []).map(r => ({
      ...r,
      type: r.customer_type === 'corporation' ? 'Venue' : 'Slot'
    })),
  ].sort((a, b) => new Date(b.paid_at).getTime() - new Date(a.paid_at).getTime())
   .slice(0, 8)

  setRecentBookings(all)
}
```
