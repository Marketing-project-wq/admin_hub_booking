### 3) UPDATE: `src/pages/LoginPage.jsx`

- Sekarang dapat URL param `:unit` (lewat `useParams`)
- Tampilkan label unit di atas form ("Arena Login", "Gym Login")
- Validasi unit di URL: kalau bukan 'arena'/'gym', redirect ke `/`
- `login()` sekarang pass 3 arg: email, password, **unit**
- Setelah login sukses → navigate ke `/{unit}` (bukan `/select` lagi)
- Tombol "Kembali" di atas form → navigate `/`

```jsx
import { useState } from 'react'
import { useNavigate, useParams, Link } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'

const UNIT_LABELS = { arena: 'Arena', gym: 'Gym' }

export default function LoginPage() {
  const { unit } = useParams()
  const navigate = useNavigate()
  const { login } = useAuth()

  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState('')
  const [submitting, setSubmitting] = useState(false)

  // Guard: unit harus valid
  if (!UNIT_LABELS[unit]) {
    navigate('/', { replace: true })
    return null
  }

  const handleSubmit = async (e) => {
    e.preventDefault()
    setError('')
    setSubmitting(true)
    try {
      await login(email, password, unit)
      navigate(`/${unit}`, { replace: true })
    } catch (err) {
      setError(err.message || 'Login gagal')
    } finally {
      setSubmitting(false)
    }
  }

  return (
    <div className="login-page">
      <div className="login-card">
        <Link to="/" className="back-link">← Kembali</Link>
        <h1 className="brand">20FIT</h1>
        <p className="subtitle">{UNIT_LABELS[unit]} Admin</p>

        <form onSubmit={handleSubmit}>
          <label>Email</label>
          <input
            type="email"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            required
            autoComplete="email"
          />

          <label>Password</label>
          <input
            type="password"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            required
            autoComplete="current-password"
          />

          {error && <p className="error">{error}</p>}

          <button type="submit" className="btn-primary" disabled={submitting}>
            {submitting ? 'Memproses...' : 'Masuk'}
          </button>
        </form>
      </div>
    </div>
  )
}
```

### 4) UPDATE: `src/context/AuthContext.jsx`

- `login(email, password, unit)` — accept 3 arg
- RPC call pakai signature baru: `validate_admin_login` dengan param `p_email`, `p_password`, **`p_unit`**
- Response sekarang include `unit` dan `permissions` — simpan semua di user object & localStorage

```javascript
const login = async (email, password, unit) => {
  const { data, error } = await supabase.rpc('validate_admin_login', {
    p_email: email,
    p_password: password,
    p_unit: unit,
  })
  if (error) throw error
  if (!data || data.length === 0) {
    throw new Error('Email atau password salah, atau Anda tidak memiliki akses ke unit ini')
  }
  const profile = data[0]
  // profile: { id, email, full_name, role, unit, permissions }
  localStorage.setItem('admin_user', JSON.stringify(profile))
  setUser(profile)
  return profile
}
```

const canAccessUnit = (targetUnit) => {
  if (!user) return false

  return Boolean(user.permissions?.[key])

### 5) UPDATE: `src/components/ProtectedRoute.jsx`

- Hapus `requireUnit` flag (sudah tidak relevan)
- Tambah optional `unit` prop — komponen akan validate user punya akses ke unit itu

```jsx
import { Navigate, useLocation } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'

const ProtectedRoute = ({ children, unit }) => {
  const { user, loading, canAccessUnit } = useAuth()
  const location = useLocation()

  if (loading) return <div className="loading-screen">Loading...</div>

  if (!user) {
    // Kalau ada unit target, redirect ke login unit itu, else ke landing
    return <Navigate to={unit ? `/login/${unit}` : '/'} replace />
  }

  if (unit && !canAccessUnit(unit)) {
    // User punya session tapi unit nggak match → kick balik ke landing
    return <Navigate to="/" replace />
  }

  return children
}

export default ProtectedRoute
```

### 8) UPDATE: `src/components/Sidebar.jsx`

- Accept prop `currentUnit`
- Header tampil nama unit (uppercase)
- Menu items per unit (Batch 1 cukup placeholder Dashboard)
- **Untuk super_admin:** tambah section "Switch Unit" di footer dengan 2 link (Arena & Gym), highlight yang aktif
- Tombol "Logout" tetap ada

```jsx
import { Link, useNavigate } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'

const UNIT_MENUS = {
  arena: [
    { label: 'Dashboard', path: '/arena', active: true },
    { label: 'Bookings', path: '/arena/bookings', disabled: true },
    { label: 'Vouchers', path: '/arena/vouchers', disabled: true },
    { label: 'Master Data', path: '/arena/master', disabled: true },
  ],
  gym: [
    { label: 'Dashboard', path: '/gym', active: true },
    { label: 'Bookings', path: '/gym/bookings', disabled: true },
  ],
}

const Sidebar = ({ currentUnit }) => {
  const { user, logout } = useAuth()
  const navigate = useNavigate()
  const menus = UNIT_MENUS[currentUnit] || []
  const isSuperAdmin = user?.role === 'super_admin'

  const handleLogout = () => {
    logout()
    navigate('/', { replace: true })
  }

  return (
    <aside className="sidebar">
      <div className="sidebar-header">
        <h2 className="brand-small">20FIT</h2>
        <p className="unit-label">{currentUnit?.toUpperCase()}</p>
      </div>

      <nav className="sidebar-menu">
        {menus.map((m) => (
          <Link
            key={m.path}
            to={m.path}
            className={`menu-item ${m.active ? 'active' : ''} ${m.disabled ? 'disabled' : ''}`}
            onClick={(e) => m.disabled && e.preventDefault()}
          >
            {m.label}
          </Link>
        ))}
      </nav>

      <div className="sidebar-footer">
        {isSuperAdmin && (
          <div className="unit-switcher">
            <p className="switcher-label">Switch Unit</p>
            <Link
              to="/arena"
              className={`switcher-item ${currentUnit === 'arena' ? 'active' : ''}`}
            >
              Arena
            </Link>
            <Link
              to="/gym"
              className={`switcher-item ${currentUnit === 'gym' ? 'active' : ''}`}
            >
              Gym
            </Link>
          </div>
        )}

        <div className="sidebar-user">
          <div className="sidebar-user-name">{user?.full_name}</div>
          <div className="sidebar-user-role">
            {user?.role === 'super_admin' ? 'SUPER ADMIN' : user?.role?.toUpperCase()}
          </div>
        </div>
        <button onClick={handleLogout}>Logout</button>
      </div>
    </aside>
  )
}

export default Sidebar
```
