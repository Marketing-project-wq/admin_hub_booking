## SHARED UTILITIES

### `src/lib/format.js`
```javascript
export const fmtRp = (n) => 'Rp ' + (Number(n) || 0).toLocaleString('id-ID')

export const fmtDate = (iso) => {
  if (!iso) return '-'
  return new Date(iso).toLocaleDateString('id-ID', {
    day: '2-digit', month: 'short', year: 'numeric'
  })
}

export const fmtDateTime = (iso) => {
  if (!iso) return '-'
  return new Date(iso).toLocaleString('id-ID', {
    day: '2-digit', month: 'short', year: 'numeric',
    hour: '2-digit', minute: '2-digit'
  })
}

export const fmtTime = (timeStr) => {
  if (!timeStr) return '-'
  return timeStr.substring(0, 5) // "08:00:00" → "08:00"
}

export const STATUS_LABEL = {
  confirmed:       { label: 'Confirmed',  css: 'badge-confirmed' },
  pending_payment: { label: 'Pending',    css: 'badge-pending' },
  cancelled:       { label: 'Cancelled', css: 'badge-cancelled' },
}

// Export array of objects ke CSV dan trigger download
export const exportToCSV = (data, filename) => {
  if (!data || data.length === 0) return
  const headers = Object.keys(data[0])
  const rows = data.map(row =>
    headers.map(h => {
      const val = row[h] ?? ''
      return `"${String(val).replace(/"/g, '""')}"`
    }).join(',')
  )
  const csv = [headers.join(','), ...rows].join('\n')
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = `${filename}_${new Date().toISOString().slice(0,10)}.csv`
  a.click()
  URL.revokeObjectURL(url)
}
```

Period selector: **Hari Ini / Minggu Ini / Bulan Ini / Bulan Lalu** (pill buttons, state default: Bulan Ini)

```javascript
const getPeriodRange = (period) => {
  const now = new Date()
  const start = new Date(now)
  const end = new Date(now)
  end.setHours(23, 59, 59, 999)

  if (period === 'today') {
    start.setHours(0, 0, 0, 0)
  } else if (period === 'week') {
    const day = now.getDay()
    start.setDate(now.getDate() - day)
    start.setHours(0, 0, 0, 0)
  } else if (period === 'month') {
    start.setDate(1)
    start.setHours(0, 0, 0, 0)
  } else if (period === 'last_month') {
    start.setMonth(start.getMonth() - 1, 1)
    start.setHours(0, 0, 0, 0)
    end.setDate(0) // last day of prev month
    end.setHours(23, 59, 59, 999)
  }
  return { start: start.toISOString(), end: end.toISOString() }
}
```

## POLA UMUM IMPLEMENTASI

Setiap page mengikuti pola ini (pseudocode):

```javascript
const [data, setData] = useState([])
const [loading, setLoading] = useState(true)
const [search, setSearch] = useState('')
const [statusFilter, setStatusFilter] = useState('all')
const [page, setPage] = useState(0)
const PAGE_SIZE = 20

// Fetch dengan filter
const fetchData = async () => {
  setLoading(true)
  let query = supabase.from('arena_bookings').select('*, unit:arena_booking_units(name)', { count: 'exact' })
  if (statusFilter !== 'all') query = query.eq('status', statusFilter)
  if (search) query = query.or(`full_name.ilike.%${search}%,booking_code.ilike.%${search}%`)
  query = query.order('created_at', { ascending: false }).range(page * PAGE_SIZE, (page + 1) * PAGE_SIZE - 1)
  const { data, count, error } = await query
  if (!error) { setData(data); setTotal(count) }
  setLoading(false)
}

// Re-fetch saat filter / page berubah
useEffect(() => { fetchData() }, [statusFilter, search, page])

// Debounce search input (300ms) supaya tidak flood request
```

**CATATAN PENTING:**
- Semua angka Rupiah: `fmtRp()` dari `format.js`
- Semua tanggal: `fmtDate()` atau `fmtDateTime()`
- Semua waktu (HH:MM:SS): `fmtTime()`
- Status badge: `STATUS_LABEL[status]`
- Update status selalu tambah `updated_at: new Date().toISOString()`
- Setelah mutasi (insert/update/delete): refetch data
- Error dari Supabase: tampil toast/banner merah di atas halaman
