### 3. Search debounce handler

```javascript
const handleSearchChange = (val) => {
  setSearchInput(val)
  if (searchTimer.current) clearTimeout(searchTimer.current)
  searchTimer.current = setTimeout(() => {
    setSearch(val)
    setPage(0)
  }, 300)
}
```

### 4. Reset selection saat filter berubah

```javascript
useEffect(() => {
```

### 5. Bulk cancel handler

```javascript
const handleBulkCancel = async () => {
  setBulkLoading(true)
  try {
    const ids = Array.from(selected)
    const { error } = await supabase
      .from('arena_class_schedules')
      .update({
        is_cancelled: true,
        cancelled_reason: 'Dibatalkan oleh admin',
        updated_at: new Date().toISOString(),
      })
      .in('id', ids)

    if (error) throw error
    setSelected(new Set())
    setShowBulkConfirm(false)
    fetchData()
  } catch (err) {
    setError(err.message)
  } finally {
    setBulkLoading(false)
  }
}
```

## UI — TABEL (tambah checkbox column)

### Header tabel — tambah kolom checkbox di paling kiri

```jsx
<thead>
  <tr>
    <th style={{ width: 40 }}>
      <input
        type="checkbox"
        checked={
          data.filter(r => !r.is_cancelled).length > 0 &&
          data.filter(r => !r.is_cancelled).every(r => selected.has(r.id))
        }
        onChange={e => {
          const activeIds = data.filter(r => !r.is_cancelled).map(r => r.id)
          if (e.target.checked) {
            setSelected(prev => new Set([...prev, ...activeIds]))
          } else {
            setSelected(prev => {
              const next = new Set(prev)
              activeIds.forEach(id => next.delete(id))
              return next
            })
          }
        }}
      />
    </th>
    <th>Tanggal</th>
    <th>Kelas</th>
    <th>Instruktur</th>
    <th>Waktu</th>
    <th>Quota</th>
    <th>Booked</th>
    <th>Sisa</th>
    <th>Status</th>
    <th>Aksi</th>
  </tr>
</thead>
```
