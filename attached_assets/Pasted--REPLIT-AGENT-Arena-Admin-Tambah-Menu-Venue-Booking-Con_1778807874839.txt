### Fetch bookings

```javascript
const fetchBookings = async () => {
  setLoading(true)
  let query = supabase
    .from('arena_bookings')
    .select('*', { count: 'exact' })
    .order('created_at', { ascending: false })
    .range(page * PAGE_SIZE, (page + 1) * PAGE_SIZE - 1)

  if (statusFilter !== 'all') query = query.eq('status', statusFilter)
  if (search) query = query.or(`full_name.ilike.%${search}%,booking_code.ilike.%${search}%,phone.ilike.%${search}%`)
  if (dateFrom) query = query.gte('booking_date', dateFrom)
  if (dateTo) query = query.lte('booking_date', dateTo)

  const { data, count, error } = await query
  if (!error) { setBookings(data || []); setTotal(count || 0) }
  setLoading(false)
}

useEffect(() => { fetchBookings() }, [statusFilter, search, dateFrom, dateTo, page])
```

### Form Booking Baru (Modal)

**Field-field:**

```jsx
// INFORMASI PELANGGAN
Nama / Organisasi*  [text input]          → full_name (NOT NULL)
Email*              [email input]         → email (NOT NULL, fallback 'noemail@20fit.id' jika kosong)
No. Telepon*        [text input]          → phone (NOT NULL)
Tipe Customer       [radio: Individu / Korporasi]  → customer_type: 'individual'/'corporation'

// DETAIL VENUE
Tanggal*            [date input, min=today]  → booking_date
Jam Mulai*          [time input]             → start_time
Jam Selesai*        [time input]             → end_time
Catatan             [textarea, optional]      → notes

// HARGA
Harga Normal (Rp)*  [number input]           → price_before_disc
Diskon (Rp)         [number input, default 0] → discount
Harga Final (Rp)    [readonly, calculated]    → price = price_before_disc - discount

// PEMBAYARAN
Status Pembayaran   [radio: Belum Bayar / Sudah Bayar]
                    → Belum Bayar: status='pending_payment', paid_at=null
                    → Sudah Bayar: status='confirmed', paid_at=NOW()
Referensi Transfer  [text, tampil hanya jika Sudah Bayar, optional] → payment_ref
```

**Submit handler:**

```javascript
const handleCreate = async () => {
  // Validasi
  if (!form.full_name || !form.booking_date || !form.start_time || !form.end_time || !form.price_before_disc) {
    setFormError('Lengkapi semua field wajib')
    return
  }
  if (form.start_time >= form.end_time) {
    setFormError('Jam selesai harus lebih dari jam mulai')
    return
  }

  setSubmitting(true)
  try {
    // Generate booking code
    const { data: codeData } = await supabase.rpc('generate_booking_code')
    const bookingCode = codeData

    const priceBeforeDisc = Number(form.price_before_disc) || 0
    const discount = Number(form.discount) || 0
    const price = priceBeforeDisc - discount

    const payload = {
      booking_code:     bookingCode,
      unit_id:          '6e8f44a7-23d4-4602-90d4-980c63b3acc2',
      booking_date:     form.booking_date,
      start_time:       form.start_time,
      end_time:         form.end_time,
      booker_type:      'guest',
      customer_type:    form.customer_type || 'individual',
      full_name:        form.full_name.trim(),
      email:            form.email?.trim() || 'noemail@20fit.id',
      phone:            form.phone?.trim() || '0',
      notes:            form.notes?.trim() || null,
      price_before_disc: priceBeforeDisc,
      discount:         discount,
      price:            price,
      status:           form.paid ? 'confirmed' : 'pending_payment',
      payment_method:   'transfer',
      payment_ref:      form.payment_ref?.trim() || null,
      paid_at:          form.paid ? new Date().toISOString() : null,
    }

    const { error } = await supabase.from('arena_bookings').insert(payload)
    if (error) throw error

    setShowCreateModal(false)
    resetForm()
    fetchBookings()
  } catch (err) {
    setFormError(err.message)
  } finally {
    setSubmitting(false)
  }
}
```

const handleConfirm = async (booking) => {
  const { error } = await supabase
    .from('arena_bookings')
    .update({
      status: 'confirmed',
      payment_method: 'transfer',
      paid_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),

    .eq('id', booking.id)

  if (!error) {
    setSelectedBooking(null)
    fetchBookings()
  }

### Export CSV

Tombol Export di header (gunakan `exportToCSV` dari `src/lib/format.js`):

```javascript
const handleExport = async () => {
  const { data } = await supabase
    .from('arena_bookings')
    .select('booking_code, full_name, email, phone, customer_type, booking_date, start_time, end_time, price_before_disc, discount, price, status, payment_method, payment_ref, paid_at, notes, created_at')
    .order('booking_date', { ascending: false })

  const rows = data.map(b => ({
    ...b,
    booking_date: fmtDate(b.booking_date),
    start_time: fmtTime(b.start_time),
    end_time: fmtTime(b.end_time),
    paid_at: fmtDateTime(b.paid_at),
    created_at: fmtDateTime(b.created_at),
  }))

  exportToCSV(rows, 'venue_bookings')
}
```
