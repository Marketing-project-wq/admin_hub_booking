### Handler submit edit

```javascript
const [editError, setEditError] = useState('')
const [editSubmitting, setEditSubmitting] = useState(false)

const handleEditSubmit = async () => {
  setEditError('')
  const fullName  = document.getElementById('edit_full_name')?.value?.trim()
  const email     = document.getElementById('edit_email')?.value?.trim() || 'noemail@20fit.id'
  const phone     = document.getElementById('edit_phone')?.value?.trim() || '0'
  const date      = document.getElementById('edit_date')?.value
  const startTime = document.getElementById('edit_start')?.value
  const endTime   = document.getElementById('edit_end')?.value
  const custType  = document.getElementById('edit_customer_type')?.value
  const priceRaw  = Number(document.getElementById('edit_price')?.value) || 0
  const discount  = Number(document.getElementById('edit_discount')?.value) || 0
  const notes     = document.getElementById('edit_notes')?.value?.trim() || null
  const status    = document.getElementById('edit_status')?.value
  const payRef    = document.getElementById('edit_ref')?.value?.trim() || null

  if (!fullName) { setEditError('Nama wajib diisi'); return }
  if (!date)     { setEditError('Tanggal wajib diisi'); return }
  if (!startTime || !endTime) { setEditError('Jam wajib diisi'); return }
  if (startTime >= endTime)   { setEditError('Jam selesai harus lebih dari jam mulai'); return }

  const price = priceRaw - discount

  setEditSubmitting(true)
  const { error } = await supabase
    .from('arena_bookings')
    .update({
      full_name:       fullName,
      email:           email,
      phone:           phone,
      booking_date:    date,
      start_time:      startTime,
      end_time:        endTime,
      customer_type:   custType,
      price_before_disc: priceRaw,
      discount:        discount,
      price:           price,
      notes:           notes,
      status:          status,
      payment_ref:     payRef,
      payment_method:  status === 'confirmed' ? 'transfer' : null,
      paid_at:         status === 'confirmed' ? (editBooking.paid_at || new Date().toISOString()) : null,
      updated_at:      new Date().toISOString(),
    })
    .eq('id', editBooking.id)

  if (error) {
    setEditError(error.message)
  } else {
    setEditBooking(null)
    fetchBookings()
  }
  setEditSubmitting(false)
}
```

## UPDATE 2 — CLASS BOOKINGS: Kolom Paid At + Sort + Filter Range

### 2b. Update fetch query

```javascript
const fetchData = useCallback(async () => {
  setLoading(true)
  let query = supabase
    .from('arena_class_bookings')
    .select(`
      id, booking_code, schedule_id, booker_type, customer_type,
      full_name, email, phone, notes, price, discount, price_before_disc,
      status, payment_method, payment_ref, paid_at, created_at, updated_at,
      schedule:arena_class_schedules(
        schedule_date, start_time, end_time, instructor,
        class_type:arena_class_types(name, color)
      )
    `, { count: 'exact' })

  // Status filter
  if (statusFilter !== 'all') query = query.eq('status', statusFilter)

  // Search
  if (search) query = query.or(`full_name.ilike.%${search}%,booking_code.ilike.%${search}%`)

  // Date range filter
  if (dateFrom) {
    if (filterType === 'schedule_date') {
      // schedule_date filter butuh join — filter client-side setelah fetch
      // (Supabase tidak support filter pada nested select langsung)
    } else {
      query = query.gte(filterType, dateFrom + 'T00:00:00')
    }
  }
  if (dateTo) {
    if (filterType !== 'schedule_date') {
      query = query.lte(filterType, dateTo + 'T23:59:59')
    }
  }

  // Sort: default paid_at DESC (terbaru di atas), null paid_at di bawah
  query = query
    .order('paid_at',     { ascending: false, nullsFirst: false })
    .order('created_at',  { ascending: false })
    .range(page * PAGE_SIZE, (page + 1) * PAGE_SIZE - 1)

  const { data: rows, count, error: err } = await query
  if (err) setError(err.message)
  else {
    let filtered = rows || []

    // Client-side filter untuk schedule_date
    if (filterType === 'schedule_date' && (dateFrom || dateTo)) {
      filtered = filtered.filter(r => {
        const sd = r.schedule?.schedule_date
        if (!sd) return false
        if (dateFrom && sd < dateFrom) return false
        if (dateTo   && sd > dateTo)   return false
        return true
      })
    }

    setData(filtered)
    setTotal(filterType === 'schedule_date' ? filtered.length : (count || 0))
  }
  setLoading(false)
}, [search, statusFilter, page, filterType, dateFrom, dateTo])
```
